📉 Calculate and Plot Daily Returns¶

In [13]:
# Calculate daily return
df['Daily Return'] = df['Close'].pct_change()
df['Daily Return'].plot(figsize=(12,5), legend=True, linestyle='--', marker='o', title='Daily Returns')
plt.grid(True)
plt.show()

📊 Calculate 20-day and 50-day Moving Averages¶

In [15]:
df['MA20'] = df['Close'].rolling(window=20).mean()
df['MA50'] = df['Close'].rolling(window=50).mean()
df[['Close', 'MA20', 'MA50']].plot(figsize=(14, 6), title='Stock Price with Moving Averages')
plt.grid(True)
plt.show()
